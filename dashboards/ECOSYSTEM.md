# ═══════════════════════════════════════════════════════════════════════════
# KEY MANAGEMENT
~/launchpad/vault/deploy-keys.sh --list              # List all projects
~/launchpad/vault/deploy-keys.sh --info sftwrks      # Show project keys

# ═══════════════════════════════════════════════════════════════════════════
